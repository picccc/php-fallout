CREATE TABLE characters(id INT NOT NULL AUTO_INCREMENT NOT NULL, Name VARCHAR(20) NOT NULL default 'no_name', Surname VARCHAR(20) NOT NULL default 'no_surname', Age TINYINT(2) NOT NULL default '00', Sex ENUM('Male','Female','Unknown') NOT NULL default 'Unknown', Strength TINYINT(2) NOT NULL default '00', Perception TINYINT(2) NOT NULL default '00', Endurance TINYINT(2) NOT NULL default '00', Charisma TINYINT(2) NOT NULL default '00', Intelligence TINYINT(2) NOT NULL default '00', Agility TINYINT(2) NOT NULL default '00', Luck TINYINT(2) NOT NULL default '00', PRIMARY KEY (ID));
CREATE TABLE weapons(id int NOT NULL AUTO_INCREMENT, Name varchar(200) NOT NULL default 'no_name', Type ENUM("Small","Big","Energy","Explosives","Melee","Unarmed") NOT NULL default "Unarmed", Subtype ENUM("Pistol","Rifle","Shotgun","Submachine Gun","Direct-Fire","Area-Of-Effect","Energy Pistol","Energy Rifle","Placed","Thrown","Bladed","Blunt") NOT NULL default 'Blunt', Damage int(5) NOT NULL default 0, Critical int(5) NOT NULL default 0, capacity int(5) NOT NULL default 0, weight int(5) NOT NULL default 0 , PRIMARY KEY (ID));
CREATE TABLE char_gun(char_id INT, gun_id INT);
